
DECLARE
CURSOR C_1 IS
SELECT * FROM EMPLOYEES;
X C_1%ROWTYPE;
BEGIN
OPEN C_1;
LOOP
FETCH C_1 INTO X;
--DBMS_OUTPUT.PUT_LINE(x.employee_id);
exit when c_1%notfound;
insert INTO TA_1 VALUES(X.EMPLOYEE_ID,X.SALARY,X.COMMISSION_PCT);
UPDATE TA_1 SET SALARY=0 WHERE EMPLOYEE_ID=X.EMPLOYEE_ID AND X.COMMISSION_PCT IS NULL;
end loop;
end;
